def solve(
        kitchen_length: int = 12,  # 12 foot by 16 foot kitchen
        kitchen_width: int = 16,   # 12 foot by 16 foot kitchen
        ceiling_height: int = 10,  # 10 foot high ceilings
        coats_of_primer: int = 1,  # one coat of primer
        coats_of_paint: int = 2,   # two coats of paint
        painting_rate: int = 40    # can paint 40 square feet per hour
):
    """Index: 3518.
    Returns: the number of hours needed to paint the kitchen.
    """
    #: L1
    wall_12_area = kitchen_length * ceiling_height

    #: L2
    total_12_wall_area = wall_12_area * 2

    #: L3
    wall_16_area = kitchen_width * ceiling_height

    #: L4
    total_16_wall_area = wall_16_area * 2

    #: L5
    total_wall_area = total_12_wall_area + total_16_wall_area

    #: L6
    total_coats_area = total_wall_area * (coats_of_primer + coats_of_paint)

    #: L7
    hours_to_paint = total_coats_area / painting_rate

    answer = hours_to_paint  # FINAL ANSWER
    return answer
