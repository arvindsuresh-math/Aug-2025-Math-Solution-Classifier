def solve(
    total_skeletons: int = 20, # there are 20 skeletons
    bones_adult_woman: int = 20, # an adult woman has 20 bones in their body
    bones_more_male: int = 5 # a male has 5 more than this
):
    """Index: 1205.
    Returns: the total number of bones in the graveyard.
    """
    #: L1
    num_adult_women = total_skeletons // 2

    #: L2
    remaining = total_skeletons - num_adult_women
    num_adult_men = remaining // 2
    num_children = remaining // 2

    #: L3
    total_bones_women = bones_adult_woman * num_adult_women

    #: L4
    bones_adult_man = bones_adult_woman + bones_more_male

    #: L5
    total_bones_men = bones_adult_man * num_adult_men

    #: L6
    bones_child = bones_adult_woman // 2

    #: L7
    total_bones_children = num_children * bones_child

    #: L8
    answer = total_bones_children + total_bones_men + total_bones_women # FINAL ANSWER
    return answer
