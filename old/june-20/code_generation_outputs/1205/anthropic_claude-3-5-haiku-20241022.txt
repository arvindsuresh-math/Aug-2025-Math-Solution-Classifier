def solve(
    total_skeletons: int = 20,  # In a graveyard, there are 20 skeletons
    bones_adult_woman: int = 20,  # An adult woman has 20 bones in their body
    bones_extra_for_man: int = 5,  # A male has 5 more bones than an adult woman
    bones_child_fraction: float = 0.5  # A child has half as many bones as an adult woman
):
    """Index: 1205.
    Returns: the total number of bones in the graveyard.
    """
    #: L1
    num_adult_women = total_skeletons / 2

    #: L2
    num_adult_men = num_adult_women / 2
    num_children = num_adult_women / 2

    #: L3
    bones_adult_women = num_adult_women * bones_adult_woman

    #: L4
    bones_per_adult_man = bones_adult_woman + bones_extra_for_man

    #: L5
    bones_adult_men = num_adult_men * bones_per_adult_man

    #: L6
    bones_per_child = bones_adult_woman * bones_child_fraction

    #: L7
    bones_children = num_children * bones_per_child

    #: L8
    total_bones = bones_children + bones_adult_men + bones_adult_women

    answer = total_bones  # FINAL ANSWER
    return answer
