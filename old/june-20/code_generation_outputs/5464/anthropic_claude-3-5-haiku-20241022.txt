def solve(
        total_milk: int = 16,  # Bill got 16 gallons of milk
        sour_cream_fraction: float = 1/4,  # turned 1/4 into sour cream
        butter_fraction: float = 1/4,  # turned 1/4 into butter
        milk_to_butter_ratio: int = 4,  # 4 gallons of milk to make one gallon of butter
        milk_to_sour_cream_ratio: int = 2,  # 2 gallons of milk to make 1 gallon of sour cream
        butter_price: int = 5,  # butter sells for $5/gallon
        sour_cream_price: int = 6,  # sour cream sells for $6/gallon
        whole_milk_price: int = 3  # whole milk sells for $3/gallon
    ):
    """Index: 5464.
    Returns: the total money Bill makes from selling milk products.
    """
    #: L1
    milk_for_products = total_milk * (sour_cream_fraction + butter_fraction)

    #: L2
    butter_gallons = milk_for_products / milk_to_butter_ratio

    #: L3
    sour_cream_gallons = milk_for_products / milk_to_sour_cream_ratio

    #: L4
    whole_milk_gallons = total_milk - milk_for_products

    #: L5
    whole_milk_revenue = whole_milk_gallons * whole_milk_price

    #: L6
    sour_cream_revenue = sour_cream_gallons * sour_cream_price

    #: L7
    total_revenue = whole_milk_revenue + sour_cream_revenue + (butter_gallons * butter_price)

    answer = total_revenue  # FINAL ANSWER
    return answer
