def solve(
    total_milk_gallons: int = 16, # Bill milked his cow and got 16 gallons of milk
    fraction_sour_cream: float = 1/4, # he turned 1/4 into sour cream
    fraction_butter: float = 1/4, # 1/4 into butter
    milk_per_gallon_butter: int = 4, # it takes 4 gallons of milk to make one gallon of butter
    milk_per_gallon_sour_cream: int = 2, # 2 gallons of milk to make 1 gallon of sour cream
    butter_price_per_gallon: int = 5, # Bill sells butter for $5/gallon
    sour_cream_price_per_gallon: int = 6, # sour cream for $6/gallon
    whole_milk_price_per_gallon: int = 3 # whole milk for $3/gallon
):
    """Index: 5464.
    Returns: the total amount of money Bill earns from selling butter, sour cream, and whole milk.
    """
    #: L1
    milk_for_sour_cream = total_milk_gallons * fraction_sour_cream
    milk_for_butter = total_milk_gallons * fraction_butter

    #: L2
    butter_gallons = milk_for_butter / milk_per_gallon_butter

    #: L3
    sour_cream_gallons = milk_for_sour_cream / milk_per_gallon_sour_cream

    #: L4
    whole_milk_gallons = total_milk_gallons - milk_for_butter - milk_for_sour_cream

    #: L5
    whole_milk_revenue = whole_milk_gallons * whole_milk_price_per_gallon

    #: L6
    sour_cream_revenue = sour_cream_gallons * sour_cream_price_per_gallon

    #: L7
    butter_revenue = butter_gallons * butter_price_per_gallon
    answer = whole_milk_revenue + sour_cream_revenue + butter_revenue # FINAL ANSWER
    return answer
