def solve(
    sam_rate_minutes: int = 10,  # Sam can assemble 1 widget every 10 minutes
    jack_and_sam_widgets: int = 2,  # When Jack helps they finish 2 widgets
    jack_and_sam_time_minutes: int = 15,  # in 15 minutes
    sam_hours_worked: int = 6,  # Sam worked for 6 hours
    jack_hours_worked: int = 4,  # Jack worked for 4 hours
    tony_hours_worked: int = 8,  # Tony worked the entire 8-hour shift
    total_widgets: int = 68  # total widgets completed at the end of the day
):
    """Index: 399.
    Returns: the time in minutes it takes Tony to assemble one widget.
    """
    #: L1
    # Sam completes a widget every 10 minutes (given)
    sam_time_per_widget = sam_rate_minutes

    #: L2
    # Jack completes 1 widget in 15 minutes (since Sam + Jack do 2 widgets in 15 minutes, and Sam does 1 in 10)
    jack_time_per_widget = jack_and_sam_time_minutes

    #: L3
    sam_widgets = sam_hours_worked * 60 / sam_time_per_widget

    #: L4
    jack_widgets = jack_hours_worked * 60 / jack_time_per_widget

    #: L5
    tony_widgets = total_widgets - sam_widgets - jack_widgets

    #: L6
    tony_time_per_widget = tony_hours_worked * 60 / tony_widgets

    answer = tony_time_per_widget  # FINAL ANSWER
    return answer
