    sam_time_per_widget: int = 10, # 1 widget every 10 minutes
    sam_jack_widgets_together: int = 2, # 2 complete widgets
    sam_jack_time_together: int = 15, # every 15 minutes
    sam_hours_worked: int = 6, # Sam worked for 6 hours
    jack_hours_worked: int = 4, # Jack was able to help out for 4 hours
    tony_hours_worked: int = 8, # Tony worked the entire 8-hour shift
    total_widgets_completed: int = 68 # completed 68 widgets
):
    """Index: 399.
    Returns: the time it takes Tony to assemble a widget in minutes.
    """

    # L2: When Jack helps they finish 2 in 15 minutes. ... So it takes Jack 15 minutes to complete a widget.
    # The solution interprets the 15 minutes from the combined rate as Jack's individual rate.
    jack_time_per_widget = sam_jack_time_together # Using the 15 minutes from the combined rate as Jack's rate as per solution logic

    #: L3
    sam_widgets = sam_hours_worked * 60 / sam_time_per_widget

    #: L4
    jack_widgets = jack_hours_worked * 60 / jack_time_per_widget

    #: L5
    tony_widgets = total_widgets_completed - sam_widgets - jack_widgets
