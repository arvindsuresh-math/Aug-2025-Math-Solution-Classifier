def solve(
    flagpole_height: int = 60  # The flagpole is 60 feet long
):
    """Index: 4670.
    Returns: the total distance the flag moved up and down the flagpole in feet."""
    
    #: L1
    half_pole_height = flagpole_height / 2

    #: L2
    total_flag_movement = flagpole_height + half_pole_height + half_pole_height + flagpole_height

    answer = total_flag_movement  # FINAL ANSWER
    return answer
