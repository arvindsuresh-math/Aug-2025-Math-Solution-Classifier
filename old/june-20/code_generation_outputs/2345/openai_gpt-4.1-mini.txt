def solve(
    peppers_per_very_spicy: int = 3,  # 3 peppers for very spicy curries
    num_very_spicy_before: int = 30,  # 30 very spicy curries before
    peppers_per_spicy: int = 2,  # 2 peppers for spicy curries
    num_spicy_before: int = 30,  # 30 spicy curries before
    peppers_per_mild: int = 1,  # 1 pepper for mild curries
    num_mild_before: int = 10,  # 10 mild curries before
    num_spicy_now: int = 15,  # 15 spicy curries now
    num_mild_now: int = 90  # 90 mild curries now
):
    """Index: 2345.
    Returns: the number of fewer peppers the curry house now buys.
    """
    #: L1
    peppers_very_spicy_before = peppers_per_very_spicy * num_very_spicy_before

    #: L2
    peppers_spicy_before = peppers_per_spicy * num_spicy_before

    #: L3
    peppers_mild_before = peppers_per_mild * num_mild_before

    #: L4
    total_peppers_before = peppers_very_spicy_before + peppers_spicy_before + peppers_mild_before

    #: L5
    peppers_spicy_now = peppers_per_spicy * num_spicy_now

    #: L6
    peppers_mild_now = peppers_per_mild * num_mild_now

    #: L7
    total_peppers_now = peppers_spicy_now + peppers_mild_now

    #: L8
    difference = total_peppers_before - total_peppers_now

    answer = difference  # FINAL ANSWER
    return answer
