def solve(
    peppers_very_spicy: int = 3, # The curry house needs 3 peppers for very spicy curries
    peppers_spicy: int = 2, # 2 peppers for spicy curries
    peppers_mild: int = 1, # and only 1 pepper for mild curries
    prev_very_spicy_curries: int = 30, # enough peppers for 30 very spicy curries
    prev_spicy_curries: int = 30, # 30 spicy curries
    prev_mild_curries: int = 10, # and 10 mild curries
    new_spicy_curries: int = 15, # enough peppers for 15 spicy curries
    new_mild_curries: int = 90 # and 90 mild curries
):
    """Index: 2345.
    Returns: the reduction in the number of peppers the curry house buys.
    """
    #: L1
    prev_peppers_for_very_spicy = peppers_very_spicy * prev_very_spicy_curries
    
    #: L2
    prev_peppers_for_spicy = peppers_spicy * prev_spicy_curries
    
    #: L3
    prev_peppers_for_mild = peppers_mild * prev_mild_curries
    
    #: L4
    total_peppers_prev = prev_peppers_for_very_spicy + prev_peppers_for_spicy + prev_peppers_for_mild
    
    #: L5
    new_peppers_for_spicy = peppers_spicy * new_spicy_curries
    
    #: L6
    new_peppers_for_mild = peppers_mild * new_mild_curries
    
    #: L7
    total_peppers_new = new_peppers_for_spicy + new_peppers_for_mild
    
    #: L8
    pepper_difference = total_peppers_prev - total_peppers_new
    
    answer = pepper_difference # FINAL ANSWER
    return answer
