def solve(
    rent1: int = 800,  # The first apartment costs $800 per month in rent
    utilities1: int = 260,  # and will cost an additional $260 per month in utilities
    rent2: int = 900,  # The second apartment costs $900 per month
    utilities2: int = 200,  # and will cost an additional $200 per month in utilities
    miles1: int = 31,  # the man would have to drive 31 miles per day for the first apartment
    miles2: int = 21,  # the man would only have to drive 21 miles per day for the second apartment
    cost_per_mile: float = 0.58,  # each mile a person drives has an average cost of 58 cents
    days: int = 20  # the man must drive to work 20 days each month
):
    """Index: 1202.
    Returns: the difference between the total monthly costs of the two apartments after factoring in utility and driving-related costs.
    """
    #: L1
    mileage_cost1 = miles1 * days * cost_per_mile

    #: L2
    total_cost1 = rent1 + utilities1 + mileage_cost1

    #: L3
    mileage_cost2 = miles2 * days * cost_per_mile

    #: L4
    total_cost2 = rent2 + utilities2 + mileage_cost2

    #: L5
    difference = total_cost1 - total_cost2

    answer = round(difference)  # FINAL ANSWER
    return answer
