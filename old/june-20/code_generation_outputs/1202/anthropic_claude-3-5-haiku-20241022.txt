def solve(
    first_apartment_rent: int = 800,  # first apartment costs $800 per month in rent
    first_apartment_utilities: int = 260,  # will cost an additional $260 per month in utilities
    first_apartment_miles: int = 31,  # would have to drive 31 miles per day
    second_apartment_rent: int = 900,  # second apartment costs $900 per month
    second_apartment_utilities: int = 200,  # will cost an additional $200 per month in utilities
    second_apartment_miles: int = 21,  # would only have to drive 21 miles
    miles_cost_per_mile: float = 0.58,  # IRS cost of 58 cents per mile
    work_days_per_month: int = 20  # must drive to work 20 days each month
):
    """Index: 1202.
    Returns: the difference in total monthly costs between the two apartments.
    """
    #: L1
    first_apartment_mileage_cost = first_apartment_miles * work_days_per_month * miles_cost_per_mile

    #: L2
    first_apartment_total_cost = first_apartment_mileage_cost + first_apartment_rent + first_apartment_utilities

    #: L3
    second_apartment_mileage_cost = second_apartment_miles * work_days_per_month * miles_cost_per_mile

    #: L4
    second_apartment_total_cost = second_apartment_mileage_cost + second_apartment_rent + second_apartment_utilities

    #: L5
    cost_difference = first_apartment_total_cost - second_apartment_total_cost

    answer = round(cost_difference)  # FINAL ANSWER
    return answer
