def solve(
    rent_apartment1: int = 800,  # first apartment costs $800 per month in rent
    utilities_apartment1: int = 260,  # additional $260 per month in utilities for first apartment
    rent_apartment2: int = 900,  # second apartment costs $900 per month
    utilities_apartment2: int = 200,  # additional $200 per month in utilities for second apartment
    miles_per_day_apartment1: int = 31,  # man drives 31 miles per day from first apartment
    miles_per_day_apartment2: int = 21,  # man drives 21 miles per day from second apartment
    days_per_month: int = 20,  # man drives to work 20 days each month
    cost_per_mile: float = 0.58  # cost per mile driven according to IRS
):
    """Index: 1202.
    Returns: the difference between the total monthly costs of the two apartments including utilities and driving costs.
    """
    #: L1
    mileage_cost_1 = miles_per_day_apartment1 * days_per_month * cost_per_mile

    #: L2
    total_cost_1 = mileage_cost_1 + rent_apartment1 + utilities_apartment1

    #: L3
    mileage_cost_2 = miles_per_day_apartment2 * days_per_month * cost_per_mile

    #: L4
    total_cost_2 = mileage_cost_2 + rent_apartment2 + utilities_apartment2

    #: L5
    difference = total_cost_1 - total_cost_2

    answer = round(difference)  # FINAL ANSWER
    return answer
