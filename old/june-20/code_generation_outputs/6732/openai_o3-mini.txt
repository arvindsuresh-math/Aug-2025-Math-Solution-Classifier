def solve(
    percent_income: float = 0.3,  # He's supposed to pay 30% of his income each year.
    years_initial: int = 3,  # For 3 years, he made $30,000/year.
    initial_salary: int = 30000,  # he made $30,000/year.
    percentage_raise: float = 20,  # then he got a 20% raise.
    years_new: int = 4,  # for the next four years.
    amount_paid: int = 1200  # he's only ever paid $1,200.
):
    """Index: 6732.
    Returns: the remaining amount owed to Nancy.
    """
    #: L1
    earnings_initial = years_initial * initial_salary

    #: L2
    raise_amount = initial_salary * percentage_raise * 0.01

    #: L3
    new_salary = initial_salary + raise_amount

    #: L4
    earnings_new = years_new * new_salary

    #: L5
    total_earnings = earnings_initial + earnings_new

    #: L6
    total_due = total_earnings * percent_income

    #: L7
    amount_owed = total_due - amount_paid

    answer = amount_owed  # FINAL ANSWER
    return answer
