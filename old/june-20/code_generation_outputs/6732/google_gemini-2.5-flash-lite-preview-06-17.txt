def solve(
    years_at_original_salary: int = 3, # For 3 years
    original_annual_salary: int = 30000, # he made $30,000/year
    raise_percentage: float = 0.20, # he got a 20% raise
    years_at_new_salary: int = 4, # for the next four years
    payment_percentage: float = 0.30, # supposed to pay 30% of his income each year
    amount_paid: int = 1200 # he's only ever paid $1,200
):
    """Index: 6732.
    Returns: the amount of child support he owes.
    """
    #: L1
    earnings_first_period = years_at_original_salary * original_annual_salary

    #: L2
    raise_amount = original_annual_salary * raise_percentage

    #: L3
    new_annual_salary = original_annual_salary + raise_amount

    #: L4
    earnings_second_period = years_at_new_salary * new_annual_salary

    #: L5
    total_earnings = earnings_first_period + earnings_second_period

    #: L6
    total_support_owed = total_earnings * payment_percentage

    #: L7
    amount_still_owed = total_support_owed - amount_paid

    answer = amount_still_owed # FINAL ANSWER
    return answer
