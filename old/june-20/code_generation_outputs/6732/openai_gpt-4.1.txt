def solve(
    percent_income_owed: float = 0.3, # He's supposed to pay 30% of his income each year
    years_at_first_salary: int = 3, # For 3 years
    first_salary: int = 30000, # he made $30,000/year
    raise_percent: float = 0.2, # then he got a 20% raise
    years_at_new_salary: int = 4, # for the next four years
    amount_paid: int = 1200 # he's only ever paid $1,200
):
    """Index: 6732.
    Returns: the total amount Nancy's ex still owes her in child support.
    """
    #: L1
    earnings_first_period = years_at_first_salary * first_salary

    #: L2
    raise_amount = first_salary * raise_percent

    #: L3
    new_salary = first_salary + raise_amount

    #: L4
    earnings_second_period = years_at_new_salary * new_salary

    #: L5
    total_earnings = earnings_first_period + earnings_second_period

    #: L6
    total_owed = total_earnings * percent_income_owed

    #: L7
    amount_owed = total_owed - amount_paid # FINAL ANSWER
    return amount_owed
