def solve(
    num_plates_initial: int = 38, # he fills a cardboard box with 38 dinner plates
    plate_weight_oz: int = 10, # each of his plates weighs 10 ounces
    box_weight_limit_lb: int = 20 # each shipping box can hold 20 pounds
):
    """Index: 3331.
    Returns: the number of plates Hasan needed to remove from the box."""
    #: L2
    box_weight_limit_oz = box_weight_limit_lb * 16  # 1 pound = 16 ounces

    #: L3
    # After removing x plates, weight is (num_plates_initial - x) * plate_weight_oz

    #: L4
    # Set up equation: (num_plates_initial - x) * plate_weight_oz = box_weight_limit_oz

    #: L5
    plates_remaining = box_weight_limit_oz // plate_weight_oz

    #: L7
    plates_removed = num_plates_initial - plates_remaining

    answer = plates_removed  # FINAL ANSWER
    return answer
