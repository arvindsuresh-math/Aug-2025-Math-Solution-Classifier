    initial_plates: int = 38,  # Hasan fills a box with 38 dinner plates

    box_weight_limit_lb: int = 20  # each shipping box can hold 20 pounds before being too heavy
):

    Returns: the number of plates Hasan needed to remove from the shipping box.
    """

    # (initial_plates - x) * plate_weight_oz = box_weight_limit_oz

    # L5-L7 combined:
