def solve(
        rise_rate: int = 50,  # balloon rises at 50 feet per minute when chain is pulled
        descent_rate: int = 10,  # balloon descends at 10 feet per minute when chain is released
        first_chain_pull_duration: int = 15,  # first chain pull lasts 15 minutes
        chain_release_duration: int = 10,  # chain is released for 10 minutes
        second_chain_pull_duration: int = 15  # second chain pull lasts 15 minutes
):
    """Index: 5918.
    Returns: the highest elevation reached by the balloon during the ride.
    """
    #: L1
    first_rise = rise_rate * first_chain_pull_duration

    #: L2
    descent = descent_rate * chain_release_duration

    #: L3
    second_rise = rise_rate * second_chain_pull_duration

    #: L4
    highest_elevation = first_rise - descent + second_rise

    answer = highest_elevation  # FINAL ANSWER
    return answer
