def solve(
        workout_goal: int = 30, # Shawna's workout goal is 30 situps
        situps_monday: int = 12, # On Monday, Shawna was only able to do 12 situps
        situps_tuesday: int = 19 # she was only able to do 19 situps on Tuesday
    ):
    """Index: 63.
    Returns: the number of situps Shawna would have to do on Wednesday to meet her goal and make up for the ones she didn't do.
    """
    #: L1
    short_monday = workout_goal - situps_monday

    #: L2
    short_tuesday = workout_goal - situps_tuesday

    #: L3
    situps_wednesday = workout_goal + short_monday + short_tuesday

    answer = situps_wednesday # FINAL ANSWER
    return answer
