def solve(
    sally_hourly_rate: int = 6, # Sally makes $6 per day
    bob_hourly_rate: int = 4, # Bob makes $4 per day
    days_in_year: int = 365 # a year has 365 days
):
    """Index: 92.
    Returns: the total amount of money Sally and Bob will have saved for their trip after a year.
    """
    #: L1
    sally_saved_per_day = sally_hourly_rate / 2

    #: L2
    sally_total_saved = sally_saved_per_day * days_in_year

    #: L3
    bob_saved_per_day = bob_hourly_rate / 2

    #: L4
    bob_total_saved = bob_saved_per_day * days_in_year

    #: L5
    total_saved = sally_total_saved + bob_total_saved

    answer = total_saved # FINAL ANSWER
    return answer
