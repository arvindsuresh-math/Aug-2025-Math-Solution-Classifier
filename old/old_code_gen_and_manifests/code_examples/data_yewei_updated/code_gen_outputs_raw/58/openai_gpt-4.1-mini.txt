def solve(
    total_students: int = 60,  # he surveyed 60 students
    fraction_receiving_6: float = 2/3,  # 2/3 of the students receive $6 allowance per day
    allowance_6: int = 6,  # average $6 allowance per day
    allowance_4: int = 4  # rest gets $4 a day
):
    """Index: 58.
    Returns: the total amount of money those 60 students get in a day.
    """
    #: L1
    students_receiving_6 = total_students * fraction_receiving_6

    #: L2
    students_receiving_4 = total_students - students_receiving_6

    #: L3
    total_allowance_6 = students_receiving_6 * allowance_6

    #: L4
    total_allowance_4 = students_receiving_4 * allowance_4

    #: L5
    total_daily_allowance = total_allowance_6 + total_allowance_4

    answer = total_daily_allowance  # FINAL ANSWER
    return answer
