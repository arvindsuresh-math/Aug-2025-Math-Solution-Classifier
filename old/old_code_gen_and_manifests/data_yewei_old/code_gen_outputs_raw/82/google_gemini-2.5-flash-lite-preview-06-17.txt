```python
# Calculate Irene's total income
regular_hours = 40
overtime_rate = 20
base_salary = 500
hours_worked = 50

# Calculate overtime hours
overtime_hours = hours_worked - regular_hours

# Calculate overtime earnings
overtime_earnings = overtime_hours * overtime_rate

# Calculate total income
total_income = base_salary + overtime_earnings

# Print the result
print(f"Irene's total income is: ${total_income}")
```
