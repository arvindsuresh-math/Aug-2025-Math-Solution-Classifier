def solve():
    # Given ratios
    elsa_ratio_part = 10
    amalie_ratio_part = 45

    # Total number of coins
    total_coins = 440

    # Fraction Amalie spends
    amalie_spend_fraction_numerator = 3

    # L2: Calculate Amalie's share of the total coins

    # L3: Calculate the number of coins Amalie spends

    # L4: Calculate the number of coins Amalie remains with
    amalie_remaining_coins = amalie_coins - coins_amalie_spends
