```python
# John's cow weight problem
def solve_87():
    initial_weight = 400
    weight_increase_factor = 1.5
    price_per_pound = 3

    # Calculate the new weight
    new_weight = initial_weight * weight_increase_factor

    # Calculate the weight gained
    weight_gained = new_weight - initial_weight

    # Calculate the increase in value
    value_increase = weight_gained * price_per_pound

    return value_increase
```
