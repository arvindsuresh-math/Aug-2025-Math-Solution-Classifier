```python
# Calculate the number of truck stamps
snowflake_stamps = 11
truck_stamps = snowflake_stamps + 9

# Calculate the number of rose stamps
rose_stamps = truck_stamps - 13

# Calculate the total number of stamps
total_stamps = snowflake_stamps + truck_stamps + rose_stamps

# Print the results
print(f"Number of snowflake stamps: {snowflake_stamps}")
print(f"Number of truck stamps: {truck_stamps}")
print(f"Number of rose stamps: {rose_stamps}")
print(f"Total number of stamps: {total_stamps}")
```
