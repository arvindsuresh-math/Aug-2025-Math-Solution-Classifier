def solve(
    fraction_saved: float = 1/2, # save half of what they've earned
    sally_daily_rate: int = 6, # Sally makes $6 per day
    bob_daily_rate: int = 4, # Bob makes $4 per day
    days_in_year: int = 365 # after a year (implied by solution)
):
    """Index: 92.
    Returns: the total amount of money Sally and Bob will have saved for their trip after a year.
    """
    #: L1
    sally_daily_savings = fraction_saved * sally_daily_rate

    #: L2
    sally_yearly_savings = sally_daily_savings * days_in_year

    #: L3
    bob_daily_savings = fraction_saved * bob_daily_rate

    #: L4
    bob_yearly_savings = bob_daily_savings * days_in_year

    #: L5
    total_savings = bob_yearly_savings + sally_yearly_savings

    answer = total_savings # FINAL ANSWER
    return answer
