def solve(
    num_daughters: int = 2, # Jerryâ€™s two daughters
    games_per_season: int = 8, # They each have 8 games this season
    practice_hours_per_game: int = 4, # Each team practices 4 hours for every game they play
    game_duration_hours: int = 2 # If each game lasts for 2 hours
):
    """Index: 60.
    Returns: the total hours Jerry spends at the field watching his daughters play and practice.
    """
    #: L1
    game_hours_one_daughter = games_per_season * game_duration_hours

    #: L2
    total_game_hours = game_hours_one_daughter * num_daughters

    #: L3
    practice_hours_one_daughter = games_per_season * practice_hours_per_game

    #: L4
    total_practice_hours = practice_hours_one_daughter * num_daughters

    #: L5
    total_hours = total_game_hours + total_practice_hours

    answer = total_hours # FINAL ANSWER
    return answer
