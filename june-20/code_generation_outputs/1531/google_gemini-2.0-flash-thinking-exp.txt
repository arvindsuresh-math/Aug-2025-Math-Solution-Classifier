def solve(
    years_working_out: int = 2, # it takes him 2 years of working out
    months_per_mountain: int = 5, # He spends 5 months climbing each mountain
    months_learning_to_dive: int = 13, # After that he takes 13 months learning to dive
    years_diving: int = 2, # dives through all the caves he wants in 2 years
    num_summits: int = 7 # climb the seven summits
):
    """Index: 1531.
    Returns: the total time taken to complete all the goals in years.
    """
    #: L1
    years_learning_climbing = years_working_out * 2

    #: L2
    months_climbing_mountains = months_per_mountain * num_summits

    #: L3
    total_months_after_climbing_and_break = months_climbing_mountains + months_learning_to_dive

    #: L4
    years_from_climbing_and_break = total_months_after_climbing_and_break / 12

    #: L5
    total_time_years = years_working_out + years_learning_climbing + years_from_climbing_and_break + years_diving

    answer = total_time_years # FINAL ANSWER
    return answer
