def solve(
        rise_rate_pulled: int = 50, # rise at a rate of 50 feet per minute
        descend_rate_released: int = 10, # descend at a rate of 10 feet per minute
        first_pull_duration: int = 15, # pulled the chain for 15 minutes
        release_duration: int = 10, # then released the rope for 10 minutes
        second_pull_duration: int = 15 # then pulled the chain for another 15 minutes
    ):
    """Index: 5918.
    Returns: the highest elevation reached by the balloon during the ride.
    """
    #: L1
    elevation_after_first_pull = rise_rate_pulled * first_pull_duration

    #: L2
    descent_during_release = descend_rate_released * release_duration

    #: L3
    elevation_gain_second_pull = rise_rate_pulled * second_pull_duration

    #: L4
    highest_elevation = elevation_after_first_pull - descent_during_release + elevation_gain_second_pull

    answer = highest_elevation # FINAL ANSWER
    return answer
