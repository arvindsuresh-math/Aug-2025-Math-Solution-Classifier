def solve(
    hours_per_chapter: float = 3, # They figure out that they should dedicate 3 hours to each chapter
    num_chapters: int = 2, # They have 2 chapters of their textbook to study
    hours_per_worksheet: float = 1.5, # and 1.5 hours for each worksheet
    num_worksheets: int = 4, # and 4 worksheets to memorize
    max_hours_per_day: int = 4, # If they plan to study no more than 4 hours each day
    break_per_hour: int = 10, # they take a 10-minute break every hour
    num_snack_breaks: int = 3, # include 3 10-minute snack breaks each day
    snack_break_duration: int = 10, # 3 10-minute snack breaks
    lunch_break_duration: int = 30 # and 30 minutes for lunch each day
):
    """Index: 7364.
    Returns: the number of days they should plan to study total over the next week."""
    #: L1
    study_hours_chapters = hours_per_chapter * num_chapters

    #: L2
    study_hours_worksheets = hours_per_chapter * num_worksheets

    #: L3
    total_study_hours = study_hours_chapters + study_hours_worksheets

    #: L4
    break_minutes_per_study_hour = total_study_hours * break_per_hour

    #: L5
    total_snack_break_minutes = num_snack_breaks * snack_break_duration

    #: L6
    total_break_minutes = break_minutes_per_study_hour + total_snack_break_minutes + lunch_break_duration
    extra_study_hours = total_break_minutes / 60

    #: L7
    total_time_needed = total_study_hours + extra_study_hours

    #: L8
    days_needed_float = total_time_needed / max_hours_per_day

    #: L9
    days_needed = int(days_needed_float) if days_needed_float == int(days_needed_float) else int(days_needed_float) + 1

    answer = days_needed # FINAL ANSWER
    return answer
