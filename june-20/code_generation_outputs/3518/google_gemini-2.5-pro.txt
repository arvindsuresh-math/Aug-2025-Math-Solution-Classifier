def solve(
    wall_length_1: int = 12, # her 12 foot by 16 foot kitchen
    wall_length_2: int = 16, # her 12 foot by 16 foot kitchen
    ceiling_height: int = 10, # has 10 foot high ceilings
    num_primer_coats: int = 1, # Each wall needs one coat of primer
    num_paint_coats: int = 2, # and two coats of paint
    painting_rate: int = 40 # Martha can paint 40 square feet per hour
):
    """Index: 3518.
    Returns: the total number of hours it will take Martha to paint the kitchen.
    """
    #: L1
    area_wall_type1 = wall_length_1 * ceiling_height

    #: L2
    total_area_walls_type1 = area_wall_type1 * 2

    #: L3
    area_wall_type2 = wall_length_2 * ceiling_height

    #: L4
    total_area_walls_type2 = area_wall_type2 * 2

    #: L5
    total_surface_area = total_area_walls_type1 + total_area_walls_type2

    #: L6
    total_area_to_paint = total_surface_area * (num_primer_coats + num_paint_coats)

    #: L7
    hours_to_paint = total_area_to_paint / painting_rate

    answer = hours_to_paint # FINAL ANSWER
    return answer
