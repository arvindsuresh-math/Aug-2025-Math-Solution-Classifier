def solve(
        total_records: int = 200,  # Peggy has 200 records
        sammy_price_per_record: int = 4,  # Sammy offers 4 dollars each
        bryan_high_price_per_record: int = 6,  # Bryan offers 6 dollars for half
        bryan_low_price_per_record: int = 1  # Bryan offers 1 dollar for other half
):
    """Index: 4483.
    Returns: the difference in profit between Sammy's and Bryan's deals."""
    
    #: L1
    sammy_total_offer = total_records * sammy_price_per_record

    #: L2
    records_per_price_point = total_records / 2

    #: L3
    bryan_high_price_total = records_per_price_point * bryan_high_price_per_record

    #: L4
    bryan_low_price_total = records_per_price_point * bryan_low_price_per_record

    #: L5
    bryan_total_offer = bryan_high_price_total + bryan_low_price_total

    #: L6
    profit_difference = sammy_total_offer - bryan_total_offer

    answer = profit_difference  # FINAL ANSWER
    return answer
