def solve(
        initial_salary: int = 30000,  # he made $30,000/year
        years_at_initial_salary: int = 3,  # For 3 years
        raise_percentage: float = 0.2,  # he got a 20% raise
        years_at_raised_salary: int = 4,  # for the next four years
        amount_paid: int = 1200,  # he's only ever paid $1,200
        child_support_percentage: float = 0.3  # supposed to pay 30% of his income
):
    """Index: 6732.
    Returns: the total amount of child support he owes Nancy.
    """
    #: L1
    earnings_first_period = initial_salary * years_at_initial_salary

    #: L2
    raise_amount = initial_salary * raise_percentage

    #: L3
    new_salary = initial_salary + raise_amount

    #: L4
    earnings_second_period = new_salary * years_at_raised_salary

    #: L5
    total_earnings = earnings_first_period + earnings_second_period

    #: L6
    total_child_support_owed = total_earnings * child_support_percentage

    #: L7
    remaining_balance = total_child_support_owed - amount_paid

    answer = remaining_balance  # FINAL ANSWER
    return answer
