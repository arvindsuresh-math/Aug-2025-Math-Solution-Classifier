def solve(
    rent_apt1: int = 800, # The first apartment costs $800 per month in rent
    utilities_apt1: int = 260, # and will cost an additional $260 per month in utilities
    miles_apt1: int = 31, # the man would have to drive 31 miles per day to get to work
    rent_apt2: int = 900, # The second apartment costs $900 per month
    utilities_apt2: int = 200, # and will cost an additional $200 per month in utilities
    miles_apt2: int = 21, # the man would only have to drive 21 miles to get to work
    cost_per_mile: float = 0.58, # each mile a person drives has an average cost of 58 cents
    days_per_month: int = 20 # the man must drive to work 20 days each month
):
    """Index: 1202.
    Returns: the difference between the total monthly costs of these two apartments after factoring in utility and driving-related costs (to the nearest whole dollar)."""
    #: L1
    mileage_cost_apt1 = miles_apt1 * days_per_month * cost_per_mile

    #: L2
    total_cost_apt1 = mileage_cost_apt1 + rent_apt1 + utilities_apt1

    #: L3
    mileage_cost_apt2 = miles_apt2 * days_per_month * cost_per_mile

    #: L4
    total_cost_apt2 = mileage_cost_apt2 + rent_apt2 + utilities_apt2

    #: L5
    difference_in_costs = total_cost_apt1 - total_cost_apt2

    answer = round(difference_in_costs) # FINAL ANSWER
    return answer
