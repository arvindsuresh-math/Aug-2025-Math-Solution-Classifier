def solve(
    keys_to_fix_initially: int = 15, # She counts 15 keys that are sticky
    time_per_key: int = 3, # times it to take 3 minutes to clean one
    assignment_time: int = 10 # Tina knows her assignment will only take 10 minutes to complete
):
    """Index: 636.
    Returns: the total time in minutes it will take for Tina to both clean the remaining keys and finish her assignment."""
    #: L1
    keys_remaining = keys_to_fix_initially - 1
    total_cleaning_time = keys_remaining * time_per_key

    #: L2
    total_time_needed = total_cleaning_time + assignment_time

    answer = total_time_needed # FINAL ANSWER
    return answer
