def solve(
    flagpole_length: int = 60  # the flagpole is 60 feet long
):
    """Index: 4670.
    Returns: the total distance the flag moved up and down the pole in feet.
    """
    #: L1
    half_mast = flagpole_length / 2

    #: L2
    total_movement = flagpole_length + half_mast + half_mast + flagpole_length

    answer = total_movement  # FINAL ANSWER
    return answer
