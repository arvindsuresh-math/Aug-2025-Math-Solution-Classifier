def solve(
    green_beads: int = 3,  # "3 green beads"
    purple_beads: int = 5,  # "5 purple beads"
    red_multiplier: int = 2,  # "twice as many red beads as green beads"
    repeats_bracelet: int = 3,  # "pattern repeats three times per bracelet"
    repeats_necklace: int = 5,  # "and 5 times per necklace"
    num_bracelets: int = 1,  # "1 bracelets"
    num_necklaces: int = 10  # "10 necklaces"
):
    """Index: 3779.
    Returns: the total number of beads needed for the bracelets and necklaces.
    """
    #: L1
    red_beads = green_beads * red_multiplier

    #: L2
    beads_per_pattern = red_beads + green_beads + purple_beads

    #: L3
    beads_per_bracelet = beads_per_pattern * repeats_bracelet

    #: L4
    beads_per_necklace = beads_per_pattern * repeats_necklace

    #: L5
    total_necklace_beads = beads_per_necklace * num_necklaces

    #: L6
    total_bracelet_beads = beads_per_bracelet * num_bracelets

    total_beads = total_necklace_beads + total_bracelet_beads

    answer = total_beads  # FINAL ANSWER
    return answer
