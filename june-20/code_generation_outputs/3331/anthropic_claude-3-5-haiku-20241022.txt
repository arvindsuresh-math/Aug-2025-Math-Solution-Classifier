def solve(
    total_plates: int = 38,  # Hasan fills a cardboard box with 38 dinner plates
    plate_weight_ounces: int = 10,  # each of his plates weighs 10 ounces
    max_box_weight_pounds: int = 20  # each shipping box can hold 20 pounds
):
    """Index: 3331.
    Returns: the number of plates Hasan needed to remove from the shipping box."""
    #: L2
    max_box_weight_ounces = max_box_weight_pounds * 16

    #: L3, L4
    # Equation: (total_plates - x) * plate_weight_ounces = max_box_weight_ounces
    # Solve for x (plates removed)

    #: L5
    x = total_plates - (max_box_weight_ounces / plate_weight_ounces)

    answer = int(x)  # FINAL ANSWER
    return answer
