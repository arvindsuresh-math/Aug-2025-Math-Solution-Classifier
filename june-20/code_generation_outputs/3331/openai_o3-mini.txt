def solve(
    num_plates: int = 38,  # "fills a cardboard box with 38 dinner plates"
    plate_weight: int = 10,  # "each of his plates weighs 10 ounces"
    weight_limit_lb: int = 20  # "each shipping box can hold 20 pounds"
):
    """Index: 3331.
    Returns: the number of plates Hasan needs to remove from the box.
    """
    #: L2
    weight_limit_oz = weight_limit_lb * 16  # 20 pounds * 16 = 320 ounces
    
    #: L3
    # Weight of the plates remaining in the box after removing x plates is (num_plates - x) * plate_weight.
    #: L4
    # Set up the equation: (num_plates - x) * plate_weight = weight_limit_oz.
    #: L5
    remaining_plates = weight_limit_oz / plate_weight  # (38 - x) = 320 / 10 = 32
    #: L6
    plates_removed = num_plates - remaining_plates  # x = 38 - 32
    #: L7
    answer = int(plates_removed)  # FINAL ANSWER
    return answer
