def solve(
    initial_plates: int = 38, # Hasan fills a cardboard box with 38 dinner plates
    weight_per_plate_oz: int = 10, # each of his plates weighs 10 ounces
    weight_limit_lb: int = 20, # each shipping box can hold 20 pounds
    lb_to_oz_conversion: int = 16 # a pound is equal to 16 ounces
):
    """Index: 3331.
    Returns: the number of plates Hasan needed to remove from the shipping box.
    """
    #: L1
    # Let x be the number of plates removed from the box. (This is what we want to find)

    #: L2
    weight_limit_oz = weight_limit_lb * lb_to_oz_conversion

    #: L3
    # The weight of the plates in the box after removing x plates is (initial_plates - x) * weight_per_plate_oz

    #: L4
    # The equation is (initial_plates - x) * weight_per_plate_oz = weight_limit_oz

    #: L5
    # Dividing both sides by weight_per_plate_oz: initial_plates - x = weight_limit_oz / weight_per_plate_oz
    plates_after_removal = weight_limit_oz / weight_per_plate_oz

    #: L6
    # Rearranging to solve for x: x = initial_plates - plates_after_removal

    #: L7
    plates_removed = initial_plates - plates_after_removal

    answer = plates_removed  # FINAL ANSWER
    return answer
