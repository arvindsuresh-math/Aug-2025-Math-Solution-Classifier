def solve(
    num_plates_initial: int = 38, # fills a cardboard box with 38 dinner plates
    plate_weight_oz: int = 10, # each of his plates weighs 10 ounces
    weight_limit_lbs: int = 20 # each shipping box can hold 20 pounds
):
    """Index: 3331.
    Returns: the number of plates Hasan needed to remove from the shipping box.
    """
    # Constant for ounces in a pound, as per L2
    ounces_per_pound = 16

    #: L2
    weight_limit_oz = weight_limit_lbs * ounces_per_pound

    # L3 and L4 describe the equation (38 - x) * 10 = 320.
    # L5 solves for the number of plates that can be in the box.
    #: L5
    max_plates_allowed = weight_limit_oz / plate_weight_oz

    # L6 and L7 solve for x, the number of plates removed.
    #: L7
    plates_removed = num_plates_initial - max_plates_allowed

    answer = plates_removed # FINAL ANSWER
    return answer
