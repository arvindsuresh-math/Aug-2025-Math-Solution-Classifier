```python
# Calculate the number of tickets eligible for discount
discount_tickets = 12 - 10

# Calculate the discount amount per ticket
discount_per_ticket = 40 * (5 / 100)

# Calculate the price of a discounted ticket
discounted_ticket_price = 40 - discount_per_ticket

# Calculate the total cost for the discounted tickets
cost_of_discounted_tickets = discounted_ticket_price * discount_tickets

# Calculate the total cost for the non-discounted tickets
cost_of_regular_tickets = 40 * 10

# Calculate the total amount Mr. Benson paid
total_paid = cost_of_regular_tickets + cost_of_discounted_tickets

print(total_paid)
```
