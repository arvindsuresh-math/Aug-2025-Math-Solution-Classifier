```python
# John writes 20 pages a day.
# He wants to write 3 books that are 400 pages each.

# Calculate the total number of pages John needs to write.
total_pages = 3 * 400

# Calculate the number of days it will take John to write the books.
days_to_write = total_pages / 20

# Print the result
print(f"It will take John {days_to_write} days to write the books.")
```
