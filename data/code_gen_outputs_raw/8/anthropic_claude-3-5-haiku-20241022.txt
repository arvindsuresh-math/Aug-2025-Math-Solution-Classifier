```python
def solve(
    budget: int = 200,  # Alexis has a budget of $200
    shirt_cost: int = 30,  # spent $30 on a button-up shirt
    pants_cost: int = 46,  # spent $46 on suit pants
    coat_cost: int = 38,  # spent $38 on a suit coat
    socks_cost: int = 11,  # spent $11 on socks
    belt_cost: int = 18,  # spent $18 on a belt
    money_left: int = 16  # has $16 left from her budget
):
    """Index: 8.
    Returns: the amount Alexis paid for the shoes."""
    
    #: L2
    total_known_expenses = shirt_cost + pants_cost + coat_cost + socks_cost + belt_cost

    #: L3
    total_spent = budget - money_left

    #: L4
    shoes_cost = total_spent - total_known_expenses

    answer = shoes_cost  # FINAL ANSWER
    return answer
```
