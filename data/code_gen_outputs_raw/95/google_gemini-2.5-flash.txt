def solve(
        num_quarters: int = 20, # Alice has 20 quarters
        iron_nickel_percentage: float = 0.20, # 20% of the nickels are iron nickels
        iron_nickel_value: int = 3 # iron nickels worth $3 each
    ):
    """Index: 95.
    Returns: the total value of Alice's money after exchanging quarters for nickels.
    """
    # Implicit values from context
    quarter_value = 0.25
    nickel_value = 0.05

    #: L1
    nickels_per_quarter = quarter_value / nickel_value

    #: L2
    total_nickels = num_quarters * nickels_per_quarter

    #: L3
    num_iron_nickels = total_nickels * iron_nickel_percentage

    #: L4
    num_regular_nickels = total_nickels - num_iron_nickels

    #: L5
    value_iron_nickels = num_iron_nickels * iron_nickel_value

    #: L6
    value_regular_nickels = num_regular_nickels * nickel_value

    #: L7
    total_value = value_iron_nickels + value_regular_nickels

    answer = total_value # FINAL ANSWER
    return answer
