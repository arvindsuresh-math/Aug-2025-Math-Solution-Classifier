        first_tank_water_gallons: int = 48, # 48 gallons of water in the first tank

        fish_eaten_in_first_tank: int = 1 # one of the first tank fish eats another
    ):

    Returns: the number of more fish in the first tank than the second tank after one fish is eaten.
    """

    second_tank_gallons = first_tank_water_gallons / first_tank_size_multiplier

    #: L2
    fish_in_second_tank = second_tank_gallons / second_tank_fish_size

    #: L3
    initial_fish_in_first_tank = first_tank_water_gallons / first_tank_fish_size

    #: L4
    final_fish_in_first_tank = initial_fish_in_first_tank - fish_eaten_in_first_tank

    #: L5
    more_fish_in_first_tank = final_fish_in_first_tank - fish_in_second_tank
