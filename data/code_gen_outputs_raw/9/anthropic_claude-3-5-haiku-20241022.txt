def solve(
        hourly_wage: float = 18.00,  # Tina makes $18.00 an hour
        hours_per_shift: int = 8,  # standard work hours per shift
        overtime_hours_per_shift: int = 2,  # overtime hours per shift (10 - 8)
        days_worked: int = 5  # she works 5 days
):
    """Index: 9.
    Returns: the total amount of money Tina makes in 5 days.
    """
    #: L1
    base_pay_per_shift = hourly_wage * hours_per_shift

    #: L2
    overtime_hours = hours_per_shift - hours_per_shift

    #: L3
    overtime_rate = hourly_wage * 0.5

    #: L4
    overtime_hourly_wage = hourly_wage + overtime_rate

    #: L5
    base_pay_total = base_pay_per_shift * days_worked

    #: L6
    overtime_pay_per_day = overtime_hourly_wage * overtime_hours_per_shift

    #: L7
    overtime_pay_total = overtime_pay_per_day * days_worked

    #: L8
    total_pay = base_pay_total + overtime_pay_total

    answer = total_pay  # FINAL ANSWER
    return answer
