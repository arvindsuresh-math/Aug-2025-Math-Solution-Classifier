def solve(
    hourly_wage: float = 18.00, # Tina makes $18.00 an hour
    regular_hours_per_shift: int = 8, # more than 8 hours per shift
    overtime_multiplier_extra: float = 1/2, # + 1/2 your hourly wage
    hours_per_day: int = 10, # works 10 hours every day
    num_days: int = 5 # for 5 days
):
    """Index: 9.
    Returns: the total amount of money Tina makes for working 10 hours a day for 5 days, including overtime pay.
    """
    #: L1
    base_pay_per_shift = regular_hours_per_shift * hourly_wage

    #: L2
    overtime_hours_per_day = hours_per_day - regular_hours_per_shift

    #: L3
    extra_overtime_pay_per_hour = hourly_wage * overtime_multiplier_extra

    #: L4
    overtime_pay_rate = hourly_wage + extra_overtime_pay_per_hour

    #: L5
    total_base_pay = base_pay_per_shift * num_days

    #: L6
    total_overtime_pay_per_day = overtime_pay_rate * overtime_hours_per_day

    #: L7
    total_overtime_pay = total_overtime_pay_per_day * num_days

    #: L8
    total_earnings = total_base_pay + total_overtime_pay

    answer = total_earnings # FINAL ANSWER
    return answer
