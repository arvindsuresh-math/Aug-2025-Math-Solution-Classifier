def solve(
    wallet_cost: int = 100, # which costs $100
    fraction_money_has: float = 1/2, # Betty has only half of the money she needs
    parents_contribution: int = 15 # Her parents decided to give her $15
):
    """Index: 2.
    Returns: the amount of money Betty still needs to buy the wallet.
    """
    #: L1
    initial_money = wallet_cost * fraction_money_has

    #: L2
    grandparents_contribution = parents_contribution * 2

    #: L3
    money_needed = wallet_cost - initial_money - grandparents_contribution - parents_contribution

    answer = money_needed # FINAL ANSWER
    return answer
