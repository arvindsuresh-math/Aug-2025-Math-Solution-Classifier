def solve(
    wallet_cost: int = 100,  # wallet costs $100
    betty_current_money: float = 0.5,  # Betty has only half of the money she needs
    parents_contribution: int = 15,  # her parents decided to give her $15
    grandparents_multiplier: int = 2  # her grandparents gave twice as much as her parents
):
    """Index: 2.
    Returns: the amount of money Betty still needs to buy the wallet.
    """
    #: L1
    betty_money = wallet_cost * betty_current_money

    #: L2
    grandparents_contribution = parents_contribution * grandparents_multiplier

    #: L3
    money_still_needed = wallet_cost - betty_money - parents_contribution - grandparents_contribution

    answer = money_still_needed  # FINAL ANSWER
    return answer
