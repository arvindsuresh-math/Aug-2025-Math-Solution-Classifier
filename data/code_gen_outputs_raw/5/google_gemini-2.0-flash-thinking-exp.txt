def solve(
    num_yellow_flowers: int = 10, # Ten of them are yellow
    percent_more_purple: int = 80, # there are 80% more of those in purple
    percent_as_many_green: int = 25 # There are only 25% as many green flowers
):
    """Index: 5.
    Returns: the total number of flowers in Mark's garden.
    """
    #: L1
    more_purple_flowers = (percent_more_purple / 100) * num_yellow_flowers

    #: L2
    num_purple_flowers = num_yellow_flowers + more_purple_flowers

    #: L3
    total_yellow_purple = num_yellow_flowers + num_purple_flowers

    #: L4
    num_green_flowers = (percent_as_many_green / 100) * total_yellow_purple

    #: L5
    total_flowers = total_yellow_purple + num_green_flowers

    answer = total_flowers # FINAL ANSWER
    return answer
