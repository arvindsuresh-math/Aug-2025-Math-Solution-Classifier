def solve(
    total_miles: int = 30, # They have one week to run 30 miles.
    days_first_three: int = 3, # On the first three days
    avg_miles_first_three_days_jesse: float = 2/3, # Jesse averages (2/3) of a mile.
    miles_day_four_jesse: int = 10, # On day four she runs 10 miles.
    avg_miles_per_day_mia_first_four: int = 3, # Mia averages 3 miles a day over the first 4 days.
    days_in_week: int = 7 # A week long race
):
    """Index: 67.
    Returns: the average of their average that they have to run over the final three days.
    """
    #: L1
    jesse_miles_first_three_days = days_first_three * avg_miles_first_three_days_jesse

    #: L2
    jesse_miles_remaining = total_miles - miles_day_four_jesse - jesse_miles_first_three_days

    #: L3
    jesse_avg_last_three_days = jesse_miles_remaining / (days_in_week - days_first_three - 1)

    #: L4
    mia_miles_first_four_days = avg_miles_per_day_mia_first_four * 4

    #: L5
    mia_miles_remaining = total_miles - mia_miles_first_four_days

    #: L6
    mia_avg_last_three_days = mia_miles_remaining / (days_in_week - 4)

    #: L7
    total_avg_last_three_days = jesse_avg_last_three_days + mia_avg_last_three_days

    #: L8
    average_of_averages = total_avg_last_three_days / 2

    answer = average_of_averages # FINAL ANSWER
    return answer
