def solve(
        num_floors: int = 10, # The hotel has 10 floors
        rooms_per_floor: int = 10 # with 10 identical rooms on each floor
    ):
    """Index: 84.
    Returns: the number of different rooms Hans could be checked into.
    """
    #: L1
    total_rooms = num_floors * rooms_per_floor

    #: L2
    unavailable_rooms = rooms_per_floor # One floor is unavailable, so the number of rooms on that floor is unavailable
    available_rooms = total_rooms - unavailable_rooms

    answer = available_rooms # FINAL ANSWER
    return answer
