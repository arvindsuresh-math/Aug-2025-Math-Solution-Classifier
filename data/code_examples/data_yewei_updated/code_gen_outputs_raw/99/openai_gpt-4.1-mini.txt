def solve(
    gallons_first_tank: int = 48,  # There are 48 gallons of water in the first tank
    size_ratio_first_to_second: int = 2,  # The first tank is twice the size of the second tank
    size_fish_second_tank: int = 2,  # two-inch fish in the second tank
    size_fish_first_tank: int = 3  # three-inch fish in the first tank
):
    """Index: 99.
    Returns: the difference in the number of fish between the first and second tank after one fish in the first tank eats another.
    """
    #: L1
    gallons_second_tank = gallons_first_tank / size_ratio_first_to_second

    #: L2
    fish_second_tank = gallons_second_tank / size_fish_second_tank

    #: L3
    fish_first_tank = gallons_first_tank / size_fish_first_tank

    #: L4
    fish_first_tank_after_eating = fish_first_tank - 1

    #: L5
    difference_fish = fish_first_tank_after_eating - fish_second_tank

    answer = difference_fish  # FINAL ANSWER
    return answer
