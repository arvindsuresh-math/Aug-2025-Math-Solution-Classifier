def solve(
        num_business_partners: int = 2, # The profit from a business transaction is shared among 2 business partners
        mike_ratio_part: int = 2, # in the ratio 2:5 respectively
        johnson_ratio_part: int = 5, # in the ratio 2:5 respectively
        johnson_got_amount: int = 2500, # If Johnson got $2500
        shirt_cost: int = 200 # on a shirt that costs $200
    ):
    """Index: 16.
    Returns: Mike's remaining share after purchasing a shirt.
    """
    #: L1
    # According to the ratio, for every 5 parts that Johnson gets, Mike gets 2 parts

    #: L2
    value_per_part = johnson_got_amount / johnson_ratio_part

    #: L3
    mike_total_share = mike_ratio_part * value_per_part

    #: L4
    mike_remaining_share = mike_total_share - shirt_cost

    answer = mike_remaining_share # FINAL ANSWER
    return answer
