def solve(
        initial_fill_proportion: float = 1/2, # She fills it halfway
        spill_proportion: float = 1/2, # spills half the water in it
        triple_factor: int = 3, # triples the amount of water
        length_ft: int = 4, # 4 feet long
        width_ft: int = 6, # 6 feet wide
        height_ft: int = 3 # 3 feet high
    ):
    """Index: 72.
    Returns: the final volume of water in the aquarium in cubic feet.
    """
    #: L1
    aquarium_volume = length_ft * width_ft * height_ft

    #: L2
    proportion_after_spill = initial_fill_proportion * spill_proportion

    #: L3
    proportion_after_refill = triple_factor * proportion_after_spill

    #: L4
    water_volume = aquarium_volume * proportion_after_refill

    answer = water_volume # FINAL ANSWER
    return answer
