def solve(
        num_houses_on_street: int = 5, # There are 5 houses on a street
        num_first_four_houses: int = 4, # each of the first four houses
        gnomes_per_house_first_four: int = 3, # has 3 gnomes in the garden
        total_gnomes_on_street: int = 20 # a total of 20 gnomes on the street
    ):
    """Index: 28.
    Returns: the number of gnomes the fifth house has.
    """
    #: L1
    gnomes_in_first_four_houses = num_first_four_houses * gnomes_per_house_first_four

    #: L2
    gnomes_in_fifth_house = total_gnomes_on_street - gnomes_in_first_four_houses

    answer = gnomes_in_fifth_house # FINAL ANSWER
    return answer
