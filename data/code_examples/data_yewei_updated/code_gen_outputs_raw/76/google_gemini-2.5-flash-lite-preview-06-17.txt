def solve(
    num_potatoes: int = 237, # A garden produced 237 potatoes
    fewer_cucumbers: int = 60, # 60 fewer cucumbers
    peppers_multiplier: int = 2 # twice as many peppers than the cucumbers
):
    """Index: 76.
    Returns: the total number of vegetables produced by the garden.
    """
    #: L1
    num_cucumbers = num_potatoes - fewer_cucumbers

    #: L2
    num_peppers = num_cucumbers * peppers_multiplier

    #: L3
    total_vegetables = num_potatoes + num_cucumbers + num_peppers

    answer = total_vegetables # FINAL ANSWER
    return answer
