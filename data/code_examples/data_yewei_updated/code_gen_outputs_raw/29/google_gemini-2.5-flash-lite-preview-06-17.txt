def solve(
    previous_rent_utility_percentage: float = 0.40, # Mrs. Snyder used to spend 40% of her monthly income on rent and utilities.
    income_increase: int = 600, # Her salary was recently increased by $600
    new_rent_utility_percentage: float = 0.25 # now her rent and utilities only amount to 25% of her monthly income.
):
    """Index: 29.
    Returns: Mrs. Snyder's previous monthly income.
    """
    # L1: Let her previous monthly income be p
    # L2: The cost of her rent and utilities was 40% of p which is (40/100)*p = 2p/5
    # L3: Her income was increased by $600 so it is now p+$600
    # L4: The cost of her rent and utilities now amount to 25% of (p+$600) which is (25/100)*(p+$600) = (p+$600)/4
    # L5: Equating both expressions for cost of rent and utilities: 2p/5 = (p+$600)/4
    # L6: Multiplying both sides of the equation by 20 gives 8p = 5p+$3000
    # L7: Subtracting 5p from both sides gives: 3p = $3000
    # L8: Dividing both sides by 3 gives p = $1000

    # We are solving for 'p' (previous monthly income) in the equation:
    # previous_rent_utility_percentage * p = new_rent_utility_percentage * (p + income_increase)
    # previous_rent_utility_percentage * p = new_rent_utility_percentage * p + new_rent_utility_percentage * income_increase
    # previous_rent_utility_percentage * p - new_rent_utility_percentage * p = new_rent_utility_percentage * income_increase
    # p * (previous_rent_utility_percentage - new_rent_utility_percentage) = new_rent_utility_percentage * income_increase
    # p = (new_rent_utility_percentage * income_increase) / (previous_rent_utility_percentage - new_rent_utility_percentage)

    # L5, L6, L7, L8 combined into a single calculation for p
    previous_monthly_income = (new_rent_utility_percentage * income_increase) / (previous_rent_utility_percentage - new_rent_utility_percentage)

    answer = previous_monthly_income # FINAL ANSWER
    return answer
