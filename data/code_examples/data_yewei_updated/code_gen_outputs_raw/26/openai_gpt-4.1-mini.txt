```python
def solve(
    liters_seawater: float = 2,  # He collects 2 liters of seawater
    salt_percentage: float = 20  # the water is 20% salt
):
    """Index: 26.
    Returns: the amount of salt in ml after all the water evaporates.
    """
    #: L1
    liters_salt = liters_seawater * salt_percentage * 0.01

    #: L2
    ml_salt = liters_salt * 1000

    answer = ml_salt  # FINAL ANSWER
    return answer
```
