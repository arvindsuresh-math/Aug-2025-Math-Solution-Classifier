def solve(
        cheddar_cheese_pounds: int = 2, # He buys 2 pounds of cheddar cheese
        cheddar_cheese_cost: int = 10, # for $10
        cream_cheese_price_factor: float = 0.5, # a pound of cream cheese that cost half the price of the cheddar cheese
        cold_cuts_price_factor: int = 2 # a pack of cold cuts that cost twice the price of the cheddar cheese
    ):
    """Index: 13.
    Returns: the total amount Jasper spends on the ingredients.
    """
    #: L1
    cream_cheese_cost = cheddar_cheese_cost * cream_cheese_price_factor

    #: L2
    cold_cuts_cost = cheddar_cheese_cost * cold_cuts_price_factor

    #: L3
    total_cost = cheddar_cheese_cost + cream_cheese_cost + cold_cuts_cost

    answer = total_cost # FINAL ANSWER
