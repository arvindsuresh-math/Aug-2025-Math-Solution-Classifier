def solve(
    craft_price: int = 12,  # Hillary sells handmade crafts for 12 dollars per craft
    crafts_sold: int = 3,   # Hillary sells 3 crafts
    extra_money: int = 7,   # is given an extra 7 dollars from an appreciative customer
    deposit_amount: int = 18  # Hillary deposits 18 dollars from today's profits
):
    """Index: 71.
    Returns: the amount of money Hillary is left with after making the deposit."""
    #: L1
    total_craft_sales = craft_price * crafts_sold

    #: L2
    total_money = total_craft_sales + extra_money

    #: L3
    money_left = total_money - deposit_amount

    answer = money_left  # FINAL ANSWER
    return answer
