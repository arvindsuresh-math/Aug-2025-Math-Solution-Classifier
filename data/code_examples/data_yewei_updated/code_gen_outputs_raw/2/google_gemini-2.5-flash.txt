def solve(
        wallet_cost: int = 100, # new wallet which costs $100
        initial_fraction_saved: float = 0.5, # Betty has only half of the money she needs
        parents_contribution: int = 15, # Her parents decided to give her $15
        grandparents_contribution_multiplier: int = 2 # her grandparents twice as much as her parents
    ):
    """Index: 2.
    Returns: how much more money Betty needs to buy the wallet.
    """
    #: L1
    initial_money_saved = wallet_cost * initial_fraction_saved

    #: L2
    grandparents_contribution = parents_contribution * grandparents_contribution_multiplier

    #: L3
    money_needed = wallet_cost - initial_money_saved - grandparents_contribution - parents_contribution

    answer = money_needed # FINAL ANSWER
    return answer
