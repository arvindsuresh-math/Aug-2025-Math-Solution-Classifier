def solve(
    wallet_cost: int = 100,  # wallet costs $100
    betty_has_fraction: float = 0.5,  # Betty has only half of the money she needs
    parents_gift: int = 15  # parents give her $15
):
    """Index: 2.
    Returns: the additional amount of money Betty needs to buy the wallet.
    """
    #: L1
    betty_initial = wallet_cost * betty_has_fraction

    #: L2
    grandparents_gift = parents_gift * 2

    #: L3
    money_needed = wallet_cost - betty_initial - grandparents_gift - parents_gift

    answer = money_needed  # FINAL ANSWER
    return answer
