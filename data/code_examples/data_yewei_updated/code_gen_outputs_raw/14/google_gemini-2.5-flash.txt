def solve(
        pages_read_in_interval: int = 8,  # Joy can read 8 pages
        time_interval_minutes: int = 20,  # in 20 minutes
        total_pages_to_read: int = 120,  # to read 120 pages
        minutes_in_an_hour: int = 60  # (Implicit: standard conversion)
    ):
    """Index: 14.
    Returns: the number of hours it will take Joy to read the specified number of pages.
    """
    #: L1
    sets_of_time_interval_in_hour = minutes_in_an_hour / time_interval_minutes

    #: L2
    pages_per_hour = pages_read_in_interval * sets_of_time_interval_in_hour

    #: L3
    hours_to_read_total = total_pages_to_read / pages_per_hour

    answer = hours_to_read_total # FINAL ANSWER
    return answer
