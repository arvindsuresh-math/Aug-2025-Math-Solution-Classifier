def solve(
    pages_read_in_20_min: int = 8, # Joy can read 8 pages of a book in 20 minutes
    time_interval_minutes: int = 20, # in 20 minutes
    total_pages_to_read: int = 120 # How many hours will it take her to read 120 pages?
):
    """Index: 14.
    Returns: the number of hours it will take Joy to read 120 pages.
    """
    #: L1
    num_intervals_in_hour = 60 / time_interval_minutes

    #: L2
    pages_read_per_hour = pages_read_in_20_min * num_intervals_in_hour

    #: L3
    hours_to_read_total_pages = total_pages_to_read / pages_read_per_hour

    answer = hours_to_read_total_pages # FINAL ANSWER
    return answer
