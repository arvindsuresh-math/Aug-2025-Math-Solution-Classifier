def solve(
        shoes_cost: int = 95, # costs $95
        saving_months: int = 3, # for the past three months
        allowance_per_month: int = 5, # gets a $5 allowance a month
        lawn_mowing_charge: int = 15, # charges $15 to mow a lawn
        driveway_shoveling_charge: int = 7, # $7 to shovel
        change_after_shoes: int = 15, # has $15 in change
        lawns_mowed: int = 4 # mows 4 lawns
    ):
    """Index: 11.
    Returns: the number of driveways Tobias shoveled.
    """
    #: L1
    total_saved = shoes_cost + change_after_shoes

    #: L2
    earned_from_allowance = saving_months * allowance_per_month

    #: L3
    earned_from_mowing_lawns = lawns_mowed * lawn_mowing_charge

    #: L4
    earned_from_shoveling_driveways = total_saved - earned_from_mowing_lawns - earned_from_allowance

    #: L5
    num_driveways_shoveled = earned_from_shoveling_driveways / driveway_shoveling_charge

    answer = num_driveways_shoveled # FINAL ANSWER
    return answer
