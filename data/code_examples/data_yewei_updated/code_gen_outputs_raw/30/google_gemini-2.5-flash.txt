def solve(
        num_people: int = 4, # Ann, Bill, Cate, and Dale each buy personal pan pizzas
        pieces_per_pizza: int = 4, # cut into 4 pieces
        num_people_eat_50_percent: int = 2, # Bill and Dale eat 50% of their pizzas
        percent_eaten_by_group1: float = 0.50, # 50% of their pizzas
        num_people_eat_75_percent: int = 2, # Ann and Cate eat 75% of the pizzas
        percent_eaten_by_group2: float = 0.75 # 75% of the pizzas
    ):
    """Index: 30.
    Returns: the number of pizza pieces left uneaten.
    """
    #: L1
    total_pizza_pieces = num_people * pieces_per_pizza

    #: L2
    pieces_eaten_by_group1 = num_people_eat_50_percent * pieces_per_pizza * percent_eaten_by_group1

    #: L3
    pieces_eaten_by_group2 = num_people_eat_75_percent * pieces_per_pizza * percent_eaten_by_group2

    #: L4
    total_pieces_eaten = pieces_eaten_by_group1 + pieces_eaten_by_group2

    #: L5
    pieces_left_uneaten = total_pizza_pieces - total_pieces_eaten

    answer = pieces_left_uneaten # FINAL ANSWER
    return answer
