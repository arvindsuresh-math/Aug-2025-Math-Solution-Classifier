def solve(
    base_pay: int = 500,  # Irene earns $500 if she works for 40 hours a week
    standard_hours: int = 40,  # standard work hours
    overtime_rate: int = 20,  # gets an extra $20 for every hour of overtime
    total_hours_worked: int = 50  # she worked 50 hours last week
):
    """Index: 82.
    Returns: Irene's total income for the week."""
    #: L1
    overtime_hours = total_hours_worked - standard_hours

    #: L2
    overtime_pay = overtime_hours * overtime_rate

    #: L3
    total_income = base_pay + overtime_pay

    answer = total_income  # FINAL ANSWER
    return answer
