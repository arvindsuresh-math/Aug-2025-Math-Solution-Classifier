def solve(
    walk_to_work_hours: int = 2,  # It takes Roque two hours to walk to work
    bike_to_work_hours: int = 1,  # It takes Roque one hour to ride his bike to work
    walk_trips_per_week: int = 3,  # Roque walks to and from work three times a week
    bike_trips_per_week: int = 2   # Roque rides his bike to and from work twice a week
):
    """Index: 18.
    Returns: the total hours Roque spends walking and biking to and from work in a week.
    """
    #: L1
    walk_hours_one_way = walk_to_work_hours * walk_trips_per_week

    #: L2
    walk_hours_round_trip = walk_hours_one_way * 2

    #: L3
    bike_hours_one_way = bike_to_work_hours * bike_trips_per_week

    #: L4
    bike_hours_round_trip = bike_hours_one_way * 2

    #: L5
    total_hours = walk_hours_round_trip + bike_hours_round_trip

    answer = total_hours  # FINAL ANSWER
    return answer
