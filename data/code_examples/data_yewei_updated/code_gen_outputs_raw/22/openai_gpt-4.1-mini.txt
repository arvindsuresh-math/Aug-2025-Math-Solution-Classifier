def solve(
    jamies_last_name_letters: int = 4,  # Jamie’s last name is Grey, which has 4 letters
    letters_removed_from_bobbie: int = 2,  # Bobbie took two letters off her last name
    difference_samantha_bobbie: int = 3  # Samantha’s last name has three fewer letters than Bobbie’s
):
    """Index: 22.
    Returns: the number of letters in Samantha’s last name.
    """
    #: L1
    bobbies_last_name_letters = jamies_last_name_letters * 2 + letters_removed_from_bobbie

    #: L2
    samanthas_last_name_letters = bobbies_last_name_letters - difference_samantha_bobbie

    answer = samanthas_last_name_letters  # FINAL ANSWER
    return answer
