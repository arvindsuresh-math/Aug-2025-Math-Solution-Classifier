```python
# John's base pay per day
base_pay_per_day = 80
# John's base workday length in hours
base_workday_hours = 8
# Performance bonus amount per day
performance_bonus = 20
# Extra hours worked for performance bonus
extra_hours_for_bonus = 2

# Calculate the total workday hours when earning the bonus
total_workday_hours = base_workday_hours + extra_hours_for_bonus

# Calculate the total daily pay when earning the bonus
total_daily_pay = base_pay_per_day + performance_bonus

# Calculate John's hourly rate when earning the bonus
hourly_rate_with_bonus = total_daily_pay / total_workday_hours

# Print the result
print(hourly_rate_with_bonus)
```
