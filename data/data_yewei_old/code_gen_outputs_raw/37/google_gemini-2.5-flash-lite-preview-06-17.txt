```python
# Calculate the cost of each item
hamburger_cost = 3 * 5
fries_cost = 1.20 * 4
soda_cost = 0.5 * 5
spaghetti_cost = 2.7

# Calculate the total bill
total_bill = hamburger_cost + fries_cost + soda_cost + spaghetti_cost

# Calculate the cost per person
cost_per_person = total_bill / 5

# Print the results
print(f"The cost of hamburgers is: ${hamburger_cost:.2f}")
print(f"The cost of French fries is: ${fries_cost:.2f}")
print(f"The cost of sodas is: ${soda_cost:.2f}")
print(f"The cost of spaghetti is: ${spaghetti_cost:.2f}")
print(f"The total bill is: ${total_bill:.2f}")
print(f"Each person will pay: ${cost_per_person:.2f}")
```
