```python
# Calculate the total ratio of coins
total_ratio = 10 + 45

# Calculate the number of coins Amalie has
amalie_coins = (45 / total_ratio) * 440

# Calculate the number of coins Amalie spends
amalie_spends = (3 / 4) * amalie_coins

# Calculate the number of coins Amalie remains with
amalie_remains = amalie_coins - amalie_spends

# Print the result
print(f"Amalie remains with {amalie_remains} coins.")
```
