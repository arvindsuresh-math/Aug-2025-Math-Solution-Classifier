def solve(
    johnson_ratio: int = 5,  # Johnson gets 5 parts in the ratio
    mike_ratio: int = 2,  # Mike gets 2 parts in the ratio
    johnson_share: int = 2500,  # Johnson got $2500
    shirt_cost: int = 200  # Mike spends $200 on a shirt
):
    """Index: 16.
    Returns: the amount of money Mike has left after buying a shirt."""
    #: L1 & L2
    share_value = johnson_share / johnson_ratio

    #: L3
    mike_share = mike_ratio * share_value

    #: L4
    mike_remaining = mike_share - shirt_cost

    answer = mike_remaining  # FINAL ANSWER
    return answer
