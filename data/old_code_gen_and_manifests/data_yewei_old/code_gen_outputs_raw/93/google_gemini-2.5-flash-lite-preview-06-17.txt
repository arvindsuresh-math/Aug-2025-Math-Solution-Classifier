```python
# Calculate the cost of beef
beef_pounds = 1000
beef_price_per_pound = 8
beef_cost = beef_pounds * beef_price_per_pound

# Calculate the amount of chicken ordered
chicken_pounds = beef_pounds * 2
chicken_price_per_pound = 3

# Calculate the cost of chicken
chicken_cost = chicken_pounds * chicken_price_per_pound

# Calculate the total cost
total_cost = beef_cost + chicken_cost

# Print the results
print(f"The cost of beef is: ${beef_cost}")
print(f"The amount of chicken ordered is: {chicken_pounds} pounds")
print(f"The cost of chicken is: ${chicken_cost}")
print(f"The total cost of everything is: ${total_cost}")
```
