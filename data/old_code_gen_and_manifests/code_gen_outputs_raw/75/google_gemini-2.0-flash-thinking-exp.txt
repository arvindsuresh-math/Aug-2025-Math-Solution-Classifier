def solve(
    cost_toast_slice: int = 1, # A slice of toast costs £1
    cost_egg: int = 3, # eggs cost £3 each
    dale_toast_slices: int = 2, # Dale had 2 slices of toast
    dale_eggs: int = 2, # and 2 eggs
    andrew_toast_slices: int = 1, # Andrew had 1 slice of toast
    andrew_eggs: int = 2 # and 2 eggs
):
    """Index: 75.
    Returns: the total cost of their breakfast.
    """
    #: L1
    cost_dale_toast = dale_toast_slices * cost_toast_slice

    #: L2
    cost_andrew_toast = andrew_toast_slices * cost_toast_slice

    #: L3
    cost_dale_eggs = dale_eggs * cost_egg

    #: L4
    cost_andrew_eggs = andrew_eggs * cost_egg

    #: L5
    total_cost = cost_dale_toast + cost_andrew_toast + cost_dale_eggs + cost_andrew_eggs

    answer = total_cost # FINAL ANSWER
    return answer
