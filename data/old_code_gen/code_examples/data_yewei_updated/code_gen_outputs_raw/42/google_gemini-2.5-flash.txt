def solve(
        num_monkeys: int = 12, # A family of 12 monkeys
        total_piles: int = 10, # collected 10 piles of bananas
        num_piles_group1: int = 6, # 6 piles
        hands_per_pile_group1: int = 9, # 9 hands
        bananas_per_hand_group1: int = 14, # 14 bananas
        hands_per_pile_group2: int = 12, # the remaining piles had 12 hands
        bananas_per_hand_group2: int = 9 # each hand having 9 bananas
    ):
    """Index: 42.
    Returns: the number of bananas each monkey would get if divided equally.
    """
    #: L1
    bananas_group1 = num_piles_group1 * hands_per_pile_group1 * bananas_per_hand_group1

    #: L2
    remaining_piles = total_piles - num_piles_group1

    #: L3
    bananas_group2 = remaining_piles * hands_per_pile_group2 * bananas_per_hand_group2

    #: L4
    total_bananas = bananas_group1 + bananas_group2

    #: L5
    bananas_per_monkey = total_bananas / num_monkeys

    answer = bananas_per_monkey # FINAL ANSWER
    return answer
