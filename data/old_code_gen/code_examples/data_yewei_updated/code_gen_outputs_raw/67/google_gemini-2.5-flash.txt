def solve(
        total_race_miles: int = 30, # They have one week to run 30 miles
        first_three_days_jesse: int = 3, # On the first three days
        jesse_avg_first_three_days: float = 2/3, # Jesse averages (2/3) of a mile
        jesse_day_four_miles: int = 10, # On day four she runs 10 miles
        mia_avg_first_four_days: int = 3, # Mia averages 3 miles a day
        first_four_days_mia: int = 4, # over the first 4 days
        final_days_count: int = 3 # over the final three days
    ):
    """Index: 67.
    Returns: the average of their average daily miles that they have to run over the final three days.
    """
    #: L1
    jesse_miles_first_three_days = first_three_days_jesse * jesse_avg_first_three_days

    #: L2
    jesse_miles_left = total_race_miles - jesse_day_four_miles - jesse_miles_first_three_days

    #: L3
    jesse_avg_final_days = jesse_miles_left / final_days_count

    #: L4
    mia_miles_first_four_days = first_four_days_mia * mia_avg_first_four_days

    #: L5
    mia_miles_left = total_race_miles - mia_miles_first_four_days

    #: L6
    mia_avg_final_days = mia_miles_left / final_days_count

    #: L7
    total_avg_final_days = jesse_avg_final_days + mia_avg_final_days

    #: L8
    average_of_their_averages = total_avg_final_days / 2

    answer = average_of_their_averages # FINAL ANSWER
    return answer
