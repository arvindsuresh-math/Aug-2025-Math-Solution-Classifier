def solve(
        initial_cans: int = 40,  # Jennifer purchased 40 cans of milk at the store
        mark_cans: int = 50,  # Mark purchased 50 cans
        jennifer_additional_cans: int = 6,  # Jennifer bought 6 additional cans
        mark_base_cans: int = 5  # for every 5 cans Mark bought
    ):
    """Index: 80.
    Returns: the total number of milk cans Jennifer brought home from the store.
    """
    #: L1
    times_additional_cans = mark_cans / mark_base_cans

    #: L2
    total_additional_cans = times_additional_cans * jennifer_additional_cans

    #: L3
    total_cans = initial_cans + total_additional_cans

    answer = total_cans  # FINAL ANSWER
    return answer
