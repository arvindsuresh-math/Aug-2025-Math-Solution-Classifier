def solve(
    mom_cup_size: int = 8, # her mom drinks an 8-ounce cup of tea
    mom_tea_ratio: int = 1, # and uses one ounce of tea
    num_people_party: int = 12, # The party has 12 people there
    party_cup_size: int = 6 # each of them wants a 6-ounce cup of tea
):
    """Index: 38.
    Returns: the total ounces of tea needed for the party.
    """
    #: L1
    total_water_ounces = num_people_party * party_cup_size

    #: L2
    total_tea_ounces = total_water_ounces / mom_cup_size * mom_tea_ratio

    answer = total_tea_ounces # FINAL ANSWER
    return answer
