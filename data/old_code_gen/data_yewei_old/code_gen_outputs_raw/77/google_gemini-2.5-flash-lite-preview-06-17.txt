# Calculate the total weight loss
weight_loss_per_month = 3
months_until_fight = 4
total_weight_loss = weight_loss_per_month * months_until_fight

# Calculate the final weight
initial_weight = 97
final_weight = initial_weight - total_weight_loss

# Print the result
print(f"The boxer will weigh {final_weight} kilograms on the day of the fight.")
