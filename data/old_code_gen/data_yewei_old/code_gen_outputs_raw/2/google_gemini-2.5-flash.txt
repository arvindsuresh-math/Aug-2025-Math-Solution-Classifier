def solve():
    # Cost of the new wallet
    wallet_cost = 100

    # Betty's initial money (half of the wallet cost)
    betty_initial_money = wallet_cost / 2

    # Money given by parents
    parents_gift = 15

    # Money given by grandparents (twice as much as parents)
    grandparents_gift = parents_gift * 2

    # Total money Betty has received or had
    total_money_betty_has = betty_initial_money + parents_gift + grandparents_gift

    # Money Betty still needs
    money_needed = wallet_cost - total_money_betty_has

    return money_needed

# Execute the function to get the answer
answer = solve()
