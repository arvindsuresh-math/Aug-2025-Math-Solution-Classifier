def solve(
        initial_rent_utilities_percentage: float = 0.40, # 40% of her monthly income on rent and utilities
        salary_increase: int = 600, # Her salary was recently increased by $600
        new_rent_utilities_percentage: float = 0.25 # now her rent and utilities only amount to 25% of her monthly income
    ):
    """Index: 29.
    Returns: her previous monthly income.
    """
    # Let 'p' be her previous monthly income.
    # The problem states that the cost of rent and utilities remains constant.

    #: L1
    # Let previous_monthly_income be 'p'

    #: L2
    # Cost of rent and utilities initially: initial_rent_utilities_percentage * p
    # This is 0.40 * p

    #: L3
    # Her new monthly income: p + salary_increase

    #: L4
    # Cost of rent and utilities now: new_rent_utilities_percentage * (p + salary_increase)
    # This is 0.25 * (p + 600)

    #: L5
    # Equating both expressions for cost of rent and utilities:
    # initial_rent_utilities_percentage * previous_monthly_income = new_rent_utilities_percentage * (previous_monthly_income + salary_increase)
    # 0.40 * previous_monthly_income = 0.25 * (previous_monthly_income + salary_increase)

    #: L6
    # Multiplying both sides by 20 (to clear decimals/fractions as per solution):
    # 0.40 * 20 * previous_monthly_income = 0.25 * 20 * (previous_monthly_income + salary_increase)
    # 8 * previous_monthly_income = 5 * (previous_monthly_income + salary_increase)
    # 8 * previous_monthly_income = 5 * previous_monthly_income + 5 * salary_increase
    right_side_constant_term = 5 * salary_increase

    #: L7
    # Subtracting 5 * previous_monthly_income from both sides:
    # 8 * previous_monthly_income - 5 * previous_monthly_income = right_side_constant_term
    # 3 * previous_monthly_income = right_side_constant_term
    coefficient_of_p = 3

    #: L8
    # Dividing both sides by 3:
    previous_monthly_income = right_side_constant_term / coefficient_of_p

    answer = previous_monthly_income # FINAL ANSWER
    return answer
